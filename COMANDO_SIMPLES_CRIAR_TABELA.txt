# Comando simples para criar tabela (copie e cole no PowerShell)

$PROJECT_ID = "monpec-sistema-rural"
$REGION = "us-central1"
$IMAGE_NAME = "gcr.io/$PROJECT_ID/monpec:latest"

gcloud run jobs delete criar-tabela --region=$REGION --quiet 2>$null

gcloud run jobs create criar-tabela `
  --region=$REGION `
  --image=$IMAGE_NAME `
  --set-env-vars="DJANGO_SETTINGS_MODULE=sistema_rural.settings_gcp,DB_NAME=monpec_db,DB_USER=monpec_user,DB_PASSWORD=L6171r12@@jjms,CLOUD_SQL_CONNECTION_NAME=monpec-sistema-rural:us-central1:monpec-db" `
  --set-cloudsql-instances=monpec-sistema-rural:us-central1:monpec-db `
  --command="python" `
  --args="manage.py,migrate,gestao_rural,0081_add_usuario_ativo" `
  --max-retries=1 `
  --memory=2Gi `
  --cpu=2 `
  --task-timeout=300

gcloud run jobs execute criar-tabela --region=$REGION --wait

gcloud run jobs delete criar-tabela --region=$REGION --quiet 2>$null
