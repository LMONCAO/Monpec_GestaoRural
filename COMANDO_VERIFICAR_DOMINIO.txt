# Comandos para verificar o domínio monpec.com.br no Google Cloud
# Execute no Cloud Shell

# Método 1: Verificar via Meta Tag (Recomendado)
# Este comando vai gerar uma meta tag que você precisa adicionar no template
gcloud domains verify monpec.com.br --web-resource

# Após adicionar a meta tag no template e fazer deploy, verifique:
gcloud domains verify monpec.com.br --web-resource --check

# Método 2: Verificar via DNS TXT Record
# Este comando vai gerar um registro TXT que você precisa adicionar no Registro.br
gcloud domains verify monpec.com.br --dns

# Após adicionar o registro TXT no Registro.br e aguardar propagação, verifique:
gcloud domains verify monpec.com.br --dns --check

# Listar domínios verificados
gcloud domains list-user-verified

# Após verificar o domínio, você pode mapeá-lo no Cloud Run:
gcloud beta run domain-mappings create \
    --service monpec \
    --domain monpec.com.br \
    --region us-central1

# Obter registros DNS após mapear:
gcloud beta run domain-mappings describe \
    --domain monpec.com.br \
    --region us-central1 \
    --format="value(status.resourceRecords)"
