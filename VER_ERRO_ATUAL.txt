# Execute este comando para ver o erro completo da última execução:

gcloud logging read "resource.type=cloud_run_job AND resource.labels.job_name=migrate-monpec AND resource.labels.location=us-central1 AND severity>=ERROR" \
    --limit 30 \
    --format="value(textPayload)" \
    --project monpec-sistema-rural | tail -50

# OU para ver o traceback completo:

gcloud logging read "resource.type=cloud_run_job AND resource.labels.job_name=migrate-monpec AND resource.labels.location=us-central1" \
    --limit 1 \
    --format="value(textPayload)" \
    --project monpec-sistema-rural | tail -100

# OU ver a última execução:

gcloud run jobs executions list --job migrate-monpec --region us-central1 --limit 1 --format="value(name)"
